from scipy.cluster.hierarchy import linkage, cut_tree
cancer2 = cancer.drop(['id'], axis = 1)
cancer3 = cancer2.drop(['diagnosis'], axis = 1)
hegits = np.arange(0, 30)
n_clusters = np.zeros(len(hegits))
linked = linkage(cancer3, metric="euclidean", method="average")

for i, d in enumerate(hegits):
    t = cut_tree(linked, height=d)
    n_clusters[i] = len(np.unique(t))

plt.plot(n_clusters, hegits, '-o')
plt.grid()
plt.xlabel('k')
plt.ylabel('heights')
